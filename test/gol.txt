class bit;

state blank #888 to zero when > is bit
	and ^> is bit
	and ^ blank;
state zero #fff is bit
	to blank when ^ is bit
		and not ^> is bit
		and not > is bit,
	to one when v one,
	to one when > one and >^ one;
state one #000 is bit
	to blank when ^ is bit
		and not ^> is bit
		and not > is bit,
	to zero when v one,
	to one when > one and >^ one,
	to zero when ^ one,
	to zero when v blank and not v< is bit;
